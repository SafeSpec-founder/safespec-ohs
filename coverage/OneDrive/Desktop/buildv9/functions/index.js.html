/**
 * Verify Firebase ID token and extract user information
 */
async function verifyToken(idToken) {
  try {
    const decodedToken = await auth.verifyIdToken(idToken);
    return decodedToken;
  } catch (error) {
    throw new HttpsError('unauthenticated', 'Invalid authentication token');
  }
}
 
/**
 * Get user role from Firestore (utility)
 */
async function fetchUserRoleFromDb(uid) {
  try {
    const userDoc = await db.collection('users').doc(uid).get();
    if (!userDoc.exists) {
      return 'user'; // Default role
    }
    return userDoc.data().role || 'user';
  } catch (error) {
    console.error('Error getting user role:', error);
    return 'user';
  }
}
 
/**
 * Get user role (callable function)
 */
export const getUserRole = onCall(async (request) => {
  const { auth } = request;
 
  if (!auth) {
    throw new HttpsError('unauthenticated', 'User must be authenticated');
  }
 
  try {
    const role = await fetchUserRoleFromDb(auth.uid);
    return { role };
  } catch (error) {
    console.error('Error getting user role:', error);
    throw new HttpsError('internal', 'Failed to get user role');
  }
});
 
/**
 * Set user role (admin only)
 */
export const setUserRole = onCall(async (request) => {
  const { auth, data } = request;
  const { targetUserId, newRole } = data;
 
  if (!auth) {
    throw new HttpsError('unauthenticated', 'User must be authenticated');
  }
 
  // Check if user has admin permissions
  const hasPermission = await checkPermission(auth.uid, 'admin');
  if (!hasPermission) {
    throw new HttpsError('permission-denied', 'Insufficient permissions');
  }
 
  // Validate role
  const validRoles = ['user', 'supervisor', 'manager', 'admin', 'super_admin'];
  if (!validRoles.includes(newRole)) {
    throw new HttpsError('invalid-argument', 'Invalid role specified');
  }
 
  try {
    await db.collection('users').doc(targetUserId).update({
      role: newRole,
      updatedAt: moment().toISOString()
    });
 
    await logAuditActivity(auth.uid, 'ROLE_CHANGED', {
      targetUserId,
      newRole
    });
 
    return { success: true };
  } catch (error) {
    console.error('Error setting user role:', error);
    throw new HttpsError('internal', 'Failed to set user role');
  }
});

/**
 * Check if user has required permission
 */
async function checkPermission(uid, requiredRole) {
  const userRole = await fetchUserRoleFromDb(uid);
  const roleHierarchy = {
    super_admin: 5,
    admin: 4,
    manager: 3,
    supervisor: 2,
    user: 1
  };
 
  const userLevel = roleHierarchy[userRole] || 0;
  const requiredLevel = roleHierarchy[requiredRole] || 0;
 
  return userLevel >= requiredLevel;
}
 
/**
 * Log audit activity
 */
async function logAuditActivity(uid, action, details = {}) {
  try {
    await db.collection('audit_logs').add({
      userId: uid,
      action,
      details,
      timestamp: moment().toISOString(),
      ip: details.ip || 'unknown',
      userAgent: details.userAgent || 'unknown'
    });
  } catch (error) {
    console.error('Error logging audit activity:', error);
  }
}

/**
 * Create user profile when a new user signs up
 */
export const createUserOnSignUp = user().onCreate(async (user) => {
  const { uid, email, displayName } = user;
 
  try {
    // Create user profile in Firestore
    await db
        .collection('users')
        .doc(uid)
        .set({
          uid,
          email,
          displayName: displayName || email.split('@')[0],
          role: 'user',
          status: 'active',
          createdAt: moment().toISOString(),
          updatedAt: moment().toISOString(),
          profile: {
            firstName: '',
            lastName: '',
            department: '',
            position: '',
            phone: '',
            emergencyContact: {
              name: '',
              phone: '',
              relationship: ''
            }
          },
          preferences: {
            notifications: {
              email: true,
              push: true,
              sms: false
            },
            language: 'en',
            timezone: 'UTC'
          },
          permissions: {
            incidents: ['read', 'create'],
            documents: ['read'],
            reports: ['read'],
            compliance: ['read']
          }
        });
 
    // Log the user creation
    await logAuditActivity(uid, 'USER_CREATED', {
      email,
      displayName
    });
 
    console.log(`User profile created for ${email}`);
  } catch (error) {
    console.error('Error creating user profile:', error);
  }
});
 
/**
 * Clean up user data when user is deleted
 */
export const cleanupUserData = user().onDelete(async (user) => {
  const { uid } = user;
 
  try {
    // Delete user profile
    await db.collection('users').doc(uid).delete();
 
    // Archive user's incidents instead of deleting
    const incidentsSnapshot = await db
        .collection('incidents')
        .where('reportedBy', '==', uid)
        .get();
 
    const batch = db.batch();
    incidentsSnapshot.docs.forEach((doc) => {
      batch.update(doc.ref, {
        reportedBy: 'DELETED_USER',
        archivedAt: moment().toISOString()
      });
    });
    await batch.commit();
 
    console.log(`User data cleaned up for UID: ${uid}`);
  } catch (error) {
    console.error('Error cleaning up user data:', error);
  }
});

function calculatePriority(severity, category) {
  const severityWeight = {
    critical: 4,
    high: 3,
    medium: 2,
    low: 1
  };
 
  const categoryWeight = {
    injury: 3,
    near_miss: 2,
    property_damage: 2,
    environmental: 2,
    security: 1
  };
 
  const score =
    (severityWeight[severity] || 1) * (categoryWeight[category] || 1);
 
  if (score >= 12) return 'critical';
  if (score >= 8) return 'high';
  if (score >= 4) return 'medium';
  return 'low';
}
 
function getDocumentChanges(before, after) {
  const changes = {};
 
  for (const key in after) {
    if (before[key] !== after[key]) {
      changes[key] = {
        from: before[key],
        to: after[key]
      };
    }
  }
 
  return changes;
}
 
function getIncidentsByCategory(incidents) {
  const categories = {};
  incidents.forEach((incident) => {
    categories[incident.category] = (categories[incident.category] || 0) + 1;
  });
  return categories;
}
 
function calculateAverageResolutionTime(incidents) {
  const resolvedIncidents = incidents.filter(
      (i) => i.status === 'resolved' && i.resolvedAt
  );
  if (resolvedIncidents.length === 0) return 0;
 
  const totalTime = resolvedIncidents.reduce((sum, incident) => {
    const created = moment(incident.createdAt);
    const resolved = moment(incident.resolvedAt);
    return sum + resolved.diff(created, 'hours');
  }, 0);
 
  return Math.round(totalTime / resolvedIncidents.length);
}

      .where('createdAt', '>=', dateRange.start)
      .where('createdAt', '<=', dateRange.end)
      .get();
 
  return snapshot.docs.map((doc) => ({ id: doc.id, ...doc.data() }));
}
 
function calculateComplianceStatus(data) {
  const total = data.length;
  const completed = data.filter((item) => item.status === 'completed').length;
  const percentage = total > 0 ? Math.round((completed / total) * 100) : 100;
 
  if (percentage >= 95) return 'excellent';
  if (percentage >= 85) return 'good';
  if (percentage >= 70) return 'fair';
  return 'poor';
}
 
async function calculateComplianceMetrics(startDate, endDate) {
  const snapshot = await db
      .collection('compliance_items')
      .where('createdAt', '>=', startDate)
      .where('createdAt', '<=', endDate)
      .get();
 
  const items = snapshot.docs.map((doc) => doc.data());
  const totalItems = items.length;
  const completedItems = items.filter(
      (item) => item.status === 'completed'
  ).length;
  const overdueItems = items.filter(
      (item) =>
        item.dueDate &&
      item.status !== 'completed' &&
      moment(item.dueDate).isBefore(endDate)
  ).length;
  const complianceRate =
    totalItems > 0 ? Math.round((completedItems / totalItems) * 100) : 100;
 
  return {
    totalItems,
    completedItems,
    overdueItems,
    complianceRate
  };
}
 
async function calculateSafetyMetrics(startDate, endDate) {
  const incidentsSnapshot = await db
      .collection('incidents')
      .where('createdAt', '>=', startDate)
      .where('createdAt', '<=', endDate)
      .get();
 
  const incidents = incidentsSnapshot.docs.map((doc) => doc.data());
  const daysWithoutIncident =
    incidents.length === 0 ? moment().diff(moment(startDate), 'days') : 0;
  const safetyScore = 100 - incidents.length * 5; // Example: lose 5 points per incident
  const riskLevel =
    safetyScore >= 90 ? 'low' : safetyScore >= 70 ? 'medium' : 'high';
 
  return {
    daysWithoutIncident,
    safetyScore: Math.max(safetyScore, 0),
    riskLevel
  };
}
 
async function generatePDFReport(data, type) {
  const pdfDoc = await PDFDocument.create();
  const page = pdfDoc.addPage([600, 800]);
  const font = await pdfDoc.embedFont(StandardFonts.Helvetica);
 
  page.drawText(`${type.toUpperCase()} REPORT`, {
    x: 50,
    y: 750,
    size: 24,
    font,
    color: rgb(0, 0, 0.8)
  });
 
  let y = 700;
  data.forEach((item, idx) => {
    page.drawText(`${idx + 1}. ${JSON.stringify(item)}`, {
      x: 50,
      y,
      size: 10,
      font
    });
    y -= 20;
  });
 
  const pdfBytes = await pdfDoc.save();
 
  // Upload to Firebase Storage
  const fileName = `reports/${type}_report_${Date.now()}.pdf`;
  const bucket = getStorage().bucket();
  const file = bucket.file(fileName);
  await file.save(Buffer.from(pdfBytes), { contentType: 'application/pdf' });
  await file.makePublic();
 
  return file.publicUrl();
}

/**

 */
export const processPendingApproval = onCall(async (request) => {
  const { auth, data } = request;
  const { approvalId, action, comments } = data;
 
  if (!auth) {
    throw new HttpsError('unauthenticated', 'User must be authenticated');
  }
 
  // Check permissions
  const hasPermission = await checkPermission(auth.uid, 'supervisor');
  if (!hasPermission) {
    throw new HttpsError('permission-denied', 'Insufficient permissions');
  }
 
  const validActions = ['approve', 'reject', 'request_changes'];
  if (!validActions.includes(action)) {
    throw new HttpsError('invalid-argument', 'Invalid action');
  }
 
  try {
    const approvalRef = db.collection('approvals').doc(approvalId);
    const approvalDoc = await approvalRef.get();
 
    if (!approvalDoc.exists) {
      throw new HttpsError('not-found', 'Approval not found');
    }
 
    const approval = approvalDoc.data();
 
    await approvalRef.update({
      status:
        action === 'approve' ?
          'approved' :
          action === 'reject' ?
          'rejected' :
          'changes_requested',
      processedBy: auth.uid,
      processedAt: moment().toISOString(),
      comments: comments || '',
      updatedAt: moment().toISOString()
    });
 
    // Update the related document/incident
    if (approval.relatedType && approval.relatedId) {
      await db
          .collection(approval.relatedType)
          .doc(approval.relatedId)
          .update({
            approvalStatus:
            action === 'approve' ?
              'approved' :
              action === 'reject' ?
              'rejected' :
              'changes_requested',
            updatedAt: moment().toISOString()
          });
    }
 
    // Create notification for the requester
    await createNotification({
      type: 'approval_processed',
      title: `Approval ${
        action === 'approve' ?
          'Approved' :
          action === 'reject' ?
          'Rejected' :
          'Changes Requested'
      }`,
      message: `Your request has been ${
        action === 'approve' ?
          'approved' :
          action === 'reject' ?
          'rejected' :
          'returned for changes'
      }`,
      targetUsers: [approval.requestedBy],
      data: { approvalId, action }
    });
 
    await logAuditActivity(auth.uid, 'APPROVAL_PROCESSED', {
      approvalId,
      action,
      relatedType: approval.relatedType,
      relatedId: approval.relatedId
    });
 
    return { success: true };
  } catch (error) {
    console.error('Error processing approval:', error);
    throw new HttpsError('internal', 'Failed to process approval');
